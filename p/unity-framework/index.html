Unity 程序基础小框架

常用教学向模块化架构：单例 → 缓存池 / 事件中心 / Mono 桥接 / 资源加载。
目标：降低耦合、复用对象、减少 GC、让非 MonoBehaviour 类也能用 Update / 协程。

单例模式基类模块

思路

  • 纯 C# 管理器：static + 泛型 BaseManager<T>

注意：继承 MonoBehaviour 的脚本不能 new，只能 AddComponent 或拖挂载。

public class BaseManager<T> where T : new()
{
    private static T _instance;

    public static T Instance
    {
        get
        {
            if (_instance == null)
                _instance = new T();
            return _instance;
        }
    }
}

使用：class EventCenter : BaseManager<EventCenter> { } → EventCenter.Instance

public class SingletonMono<T> : MonoBehaviour where T : MonoBehaviour
{
    private static T instance;

    public static T Instance => instance;

    protected virtual void Awake()
    {
        if (instance != null && instance != this)
        {
            Destroy(gameObject);
            return;
        }
        instance = this as T;
    }
}

子类重写 Awake 时需 base.Awake()。

public class SingletonMonoAuto<T> : MonoBehaviour where T : MonoBehaviour
{
    private static T instance;

    public static T Instance
    {
        get
        {
            if (instance == null)
            {
                GameObject obj = new GameObject(typeof(T).Name);
                instance = obj.AddComponent<T>();
                DontDestroyOnLoad(obj);
            }
            return instance;
        }
    }

    protected virtual void Awake()
    {
        if (instance != null && instance != this)
        {
            Destroy(gameObject);
            return;
        }
        instance = this as T;
        DontDestroyOnLoad(gameObject);
    }
}
类型 适用
BaseManager<T> EventCenter、MonoMgr、ResMgr
SingletonMono<T> 场景中预先放好一个物体
SingletonMonoAuto<T> PoolMgr 等全局自动创建

缓存池模块

作用

  • 减少反复 Instantiate / Destroy 带来的 GC 与卡顿
  • 对象停用后放入池，需要时再取出激活
  • 代价：长期占用一块内存

GC 简述

创建对象申请内存；Destroy 往往先断开引用，由 GC 在内存紧张时回收。池化可避免频繁分配。

using System.Collections.Generic;
using UnityEngine;

public class PoolMgr : SingletonMonoAuto<PoolMgr>
{
    private Dictionary<string, List<GameObject>> poolDic =
        new Dictionary<string, List<GameObject>>();

    /// <summary>从池中取物体；池空则 Resources 实例化</summary>
    public GameObject GetObj(string name)
    {
        GameObject obj = null;

        if (poolDic.TryGetValue(name, out var list) && list.Count > 0)
        {
            obj = list[0];
            list.RemoveAt(0);
        }
        else
        {
            obj = Instantiate(Resources.Load<GameObject>(name));
        }

        obj.SetActive(true);
        return obj;
    }

    /// <summary>放回池中（Deactivate）</summary>
    public void PushObj(string name, GameObject obj)
    {
        obj.SetActive(false);

        if (!poolDic.ContainsKey(name))
            poolDic[name] = new List<GameObject>();

        poolDic[name].Add(obj);
    }
}

原笔记 bug：else 分支只 Add 了空 List，没有把 obj 放进列表，已在上文修正。

using System.Collections.Generic;
using UnityEngine;
using UnityEngine.Events;

public class EventCenter : BaseManager<EventCenter>
{
    private Dictionary<string, UnityAction<object>> eventDic =
        new Dictionary<string, UnityAction<object>>();

    public void AddEventListener(string eventName, UnityAction<object> action)
    {
        if (eventDic.ContainsKey(eventName))
            eventDic[eventName] += action;
        else
            eventDic.Add(eventName, action);
    }

    public void EventTrigger(string eventName, object info = null)
    {
        if (eventDic.TryGetValue(eventName, out var action))
            action?.Invoke(info);
    }

    public void RemoveEventListener(string eventName, UnityAction<object> action)
    {
        if (!eventDic.ContainsKey(eventName)) return;

        eventDic[eventName] -= action;
        if (eventDic[eventName] == null)
            eventDic.Remove(eventName);
    }

    /// <summary>场景切换时建议清空，避免野引用</summary>
    public void Clear()
    {
        eventDic.Clear();
    }
}

原方法名 EventTigger 建议改为 EventTrigger。

public class Monster : MonoBehaviour
{
    void Start() => StartCoroutine(DeadCoroutine());

    public void Dead()
    {
        EventCenter.Instance.EventTrigger("MonsterDead", this);
    }

    IEnumerator DeadCoroutine()
    {
        yield return new WaitForSeconds(3f);
        Dead();
    }
}

顺序：若广播时还没有订阅者，回调不会执行。应先 AddEventListener 再 EventTrigger，或用延迟/初始化阶段统一注册。

public class Player : MonoBehaviour
{
    void Start()
    {
        EventCenter.Instance.AddEventListener("MonsterDead", OnMonsterDead);
    }

    void OnMonsterDead(object info)
    {
        Debug.Log("Monster dead: " + info);
    }

    void OnDestroy()
    {
        EventCenter.Instance.RemoveEventListener("MonsterDead", OnMonsterDead);
    }
}

必须在 OnDestroy 里移除监听，否则对象销毁后仍被委托引用 → 内存泄漏。

using UnityEngine;
using UnityEngine.Events;

public class MonoController : MonoBehaviour
{
    private event UnityAction updateEvent;

    void Start()
    {
        DontDestroyOnLoad(gameObject);
    }

    void Update()
    {
        updateEvent?.Invoke();
    }

    public void AddUpdateListener(UnityAction fun) => updateEvent += fun;
    public void RemoveUpdateListener(UnityAction fun) => updateEvent -= fun;
}

using System.Collections;
using UnityEngine;
using UnityEngine.Events;

public class MonoMgr : BaseManager<MonoMgr>
{
    private MonoController controller;

    public MonoMgr()
    {
        GameObject obj = new GameObject("MonoController");
        controller = obj.AddComponent<MonoController>();
        Object.DontDestroyOnLoad(obj);
    }

    public void AddUpdateListener(UnityAction fun) =>
        controller.AddUpdateListener(fun);

    public void RemoveUpdateListener(UnityAction fun) =>
        controller.RemoveUpdateListener(fun);

  public Coroutine StartCoroutine(IEnumerator routine) =>
        controller.StartCoroutine(routine);

    public void StopCoroutine(Coroutine routine) =>
        controller.StopCoroutine(routine);

    public void StopAllCoroutines() =>
        controller.StopAllCoroutines();
}

资源加载模块

概念

方式 特点
同步 Resources.Load 阻塞主线程，加载完立刻可用
异步 ResourceRequest 不卡帧，完成后在回调里使用资源

using System;
using System.Collections;
using UnityEngine;
using UnityEngine.Events;
using Object = UnityEngine.Object;

public class ResMgr : BaseManager<ResMgr>
{
    /// <summary>同步加载 Resources 下资源</summary>
    public T Load<T>(string path) where T : Object
    {
        return Resources.Load<T>(path);
    }

    /// <summary>同步加载并实例化预制体</summary>
    public GameObject LoadGameObject(string path)
    {
        GameObject prefab = Load<GameObject>(path);
        return prefab != null ? Object.Instantiate(prefab) : null;
    }

    /// <summary>异步加载资源</summary>
    public void LoadAsync<T>(string path, UnityAction<T> callback) where T : Object
    {
        MonoMgr.Instance.StartCoroutine(LoadAsyncCoroutine(path, callback));
    }

    private IEnumerator LoadAsyncCoroutine<T>(string path, UnityAction<T> callback)
        where T : Object
    {
        ResourceRequest request = Resources.LoadAsync<T>(path);
        yield return request;

        callback?.Invoke(request.asset as T);
    }

    /// <summary>异步加载并实例化</summary>
    public void LoadGameObjectAsync(string path, UnityAction<GameObject> callback)
    {
        LoadAsync<GameObject>(path, prefab =>
        {
            if (prefab != null)
                callback?.Invoke(Object.Instantiate(prefab));
            else
                callback?.Invoke(null);
        });
    }
}

模块协作与注意点

推荐初始化顺序

  1. 场景放置或自动创建 PoolMgr / MonoController（MonoMgr 构造时创建）
  2. EventCenter、ResMgr 首次访问 Instance 时创建
  3. 各玩法脚本在 Start 中注册事件、Update
  4. 场景切换：EventCenter.Instance.Clear()，按需清理池

常见问题

问题 处理
事件不触发 订阅晚于广播；检查事件名拼写
内存泄漏 OnDestroy 移除事件；池与常驻单例慎引用已销毁对象
异步资源为空 路径不含扩展名；文件在 Resources 下；等回调再使用
单例重复 Awake 里 Destroy 重复实例
Mono 双份 Update 既在自己 Update 里写逻辑，又注册 MonoMgr，避免重复
